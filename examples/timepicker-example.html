JavaScript API

// Import the component
import { PersianTimePickerElement } from '../dist/persian-timepicker-element.js';

// Get reference to the element
const timePicker = document.getElementById('time-picker-1');

// Listen for time changes
timePicker.addEventListener('timeChange', (e) => {
  console.log('Hour:', e.detail.hour);
  console.log('Minute:', e.detail.minute);
  console.log('Second:', e.detail.second);
  console.log('Formatted time:', e.detail.timeString);
  console.log('Is AM:', e.detail.isAM);
});

// Set time programmatically
timePicker.hour = 14;
timePicker.minute = 30;
timePicker.second = 0;
timePicker.updateDisplay();

// Get current time values
const currentHour = timePicker.hour;
const currentMinute = timePicker.minute;
const currentSecond = timePicker.second;
